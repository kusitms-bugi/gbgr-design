import React from "react";
import { ModeToggle } from "@gbgr/react";

## Usage (React)

```tsx
import { ModeToggle } from "@gbgr/react"
import "@gbgr/react/styles.css"

export function Example() {
  const [value, setValue] = React.useState<"light" | "dark">("light")

  return <ModeToggle value={value} onValueChange={setValue} aria-label="Color mode" />
}
```

## Demo

export function ModeToggleDemo() {
